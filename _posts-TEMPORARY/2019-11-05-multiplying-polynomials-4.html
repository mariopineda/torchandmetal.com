  1. $(2x+7)(6x-2)$
  2. $(m+9)(5m+8)$
  3. $(3x-1)^2$
  4. $(6a-b)(6a-13b)$
  5. $(2x-1)(3x-4)$
  6. $3(8x-9y)(5x-6y)$
